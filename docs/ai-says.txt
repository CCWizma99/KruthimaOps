def compute_fold_distribution_buckets(tr_rows, va_rows, test_df, key_col='conflict_key'):
    # Divide training targets into 5 equal-width structural buckets
    tr_rows['target_bin'] = pd.cut(tr_rows['transformed_target'], bins=5, labels=False)
    
    # Compute the frequency distribution count per bucket for each group
    hist_counts = tr_rows.groupby([key_col, 'target_bin']).size().unstack(fill_value=0)
    hist_pct = hist_counts.div(hist_counts.sum(axis=1), axis=0).to_dict('index')
    
    # Map the distribution shape back to train, validation, and test splits
    for df in [tr_rows, va_rows, test_df]:
        for b in range(5):
            df[f'{key_col}_hist_bucket_{b}'] = df[key_col].map(
                lambda x: hist_pct[x][b] if x in hist_pct else 0.0
            )
    return tr_rows.drop(columns=['target_bin']), va_rows, test_df
